class Contacts extends Component{

    state = {
        contacts: [...contacts.sort((a, b) => (a.lastName > b.lastName) ? 1 : -1)],
        search : '',
        genderFilter: {
            male: true,
            female: true,
            unknown: true
        }
    }

    setGenderImage(gender){
        if(gender === 'male'){
            return maleIcon;
        }
        else if(gender === 'female'){
            return femaleIcon;
        }
        else{
            return unknownIcon;
        }
    }
    
    handleGenderChange = (e) =>{
        // console.log(this.state.genderFilter)
        console.log(e.target.name,  e.target.checked)
        this.setState({
            ...this.state.genderFilter, 
            [e.target.name]: e.target.checked
        })
        // this.setState({
        //     contacts: [...contacts.filter( el => {
        //         console.log(genderFilter)
        //     })]
        // })
    }

    handleSearchChange = (e) =>{
        this.setState({
            search: e.target.value
        })

        this.setState({
            contacts:  [...contacts.filter( el => {
                return (
                el.lastName.toLowerCase().includes(e.target.value) || 
                el.firstName.toLowerCase().includes(e.target.value) || 
                el.phone.includes(e.target.value)
                )
            })]
        })
    }

    render(){
        return (
            <div className="Contacts">
                <div className='search-wrapper'>
                    <input value={this.state.search} onChange={this.handleSearchChange} placeholder='Search'/>
                </div>
                {this.state.contacts.map( (contact, i) => <Contact 
                    firstName={contact.firstName}
                    lastName={contact.lastName}
                    phone={contact.phone}
                    gender={this.setGenderImage(contact.gender)}
                    key={i}
                />)}
                <div className='checkboxes-field'>
                    <label>
                        <input type='checkbox' name='male' checked={this.state.genderFilter.male} onChange={this.handleGenderChange}/>
                        Male
                    </label>
                    <label>
                        <input type='checkbox' name='female' checked={this.state.genderFilter.female} onChange={this.handleGenderChange}/>
                        Female
                    </label>
                    <label>
                        <input type='checkbox' name='unknown' checked={this.state.genderFilter.unknown} onChange={this.handleGenderChange}/>
                        Unknown
                    </label>
                </div>
            </div>
        );
    }
}

export default Contacts;
